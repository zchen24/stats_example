F-test MATLAB

clc; clear;

% load data
load examgrades;
x = grades(:,1);
y = grades(:,2);

% test
[h, p, ci, stats] = vartest2(x, y, 'Tail', 'right');

% print answer
disp('F-Test MATLAB:');
disp(['   h = ', num2str(h)]);
disp(['   F = ', num2str(stats.fstat)]);
disp(['   p = ', num2str(p)]);

F-Test MATLAB:
   h = 1
   F = 1.7768
   p = 0.00094364

F-test MANUAL

% F value
fm = var(x)/var(y);

% F critical value
sl = 0.05;  % significance level
df1 = length(x)-1;
df2 = length(y)-1;
fcv = finv(1-sl, df1, df2);

% test
hm = (fm > fcv);

% p
pm = 1 - fcdf(fm, df1, df2);

% print answer
disp('F-Test MANUAL:');
disp(['   h = ', num2str(hm)]);
disp(['   F = ', num2str(fm)]);
disp(['   p = ', num2str(pm)]);

F-Test MANUAL:
   h = 1
   F = 1.7768
   p = 0.00094364
