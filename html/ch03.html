1 Rule

clc; clear;

% Plot -1 to 1 as white, other grey
x = -4:0.1:4;
y = normpdf(x);
area(x,y,'FaceColor',[0.8,0.8,0.8]);

hold on;
x1 = -1:0.1:1;
y1 = normpdf(x1);
area(x1,y1,'FaceColor','w');
title('1-2-3 Rule, 1 68%');
yaxis([0, 0.5]);
xlabel('Z Value');
ylabel('Prability');

% Compute the 1 68%
c = normcdf([-1 1]);
carea = c(2) - c(1);
careastr = sprintf('area 1 = %0.3f', carea);
disp(careastr);

area 1 = 0.683
